git add .         // 添加所有要提交的数据

git commit -m ""  // 提交所有的数据，并注释

git push origin linfeng  //将提交的的数据推到linfeng这个分支上（这是本地分支）

git checkout test  // 切换到test分支上

git merge linfeng  // 合并linfeng这个分支

git pull origin test  // 将test的数据拉下来

git push origin test  // 将数据推到test的分支上（这是测试分支）

git checkout linfeng   // 切换到linfeng分支上

git merge test   / 合并test这个分支  (合并后本地分支上的数据就和测试分支上的数据是一样的了)
